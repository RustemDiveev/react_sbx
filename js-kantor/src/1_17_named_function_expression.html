Именованные функциональные выражения

Существует специально для работы с рекурсией

var f = function(...) { /* function body */ }; // обычное функциональное выражение var f = function function_name(...) { /* function body */ }; // именованное функциональное выражение

function_name доступно только изнутри самой функции function_name. Такое ограничение входит в стандарт JS, и поддерживается всеми браузерами кроме IE8.

var f = function sayHi(name) { alert(sayHi); } alert(sayHi); // не видно, ошибка var test = function sayHi2(name) { sayHi2 = "something"; // в use strict - выдал бы ошибку alert(sayHi2); // перезаписать не удалось } test();

Примеры использования

Используется в первую очередь в тех ситуациях, когда функцию нужно передавать в другое место кода или перемещать из одной переменной в другую. Внутреннее имя позволяет функции надежно обращаться к самой себе, где бы она не находилась.

function f(n) { return n ? n * f(n - 1) : 1; }; alert( f(5) ); // 120 var g = f; f = null; alert( g(5) ); // запуск функции с новым именем - ошибка при выполнении var f = function factorial(n) { return n ? n * factorial(n - 1) : 1; }; var g = f; f = null; alert( g(5) ); // 120, работает
